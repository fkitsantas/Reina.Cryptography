146 private static void ValidateInput(string input, params string[] keyNames)
147 {
148 // Validate that the input string is not null or empty.
149 if (string.IsNullOrEmpty(input))
150 throw new ArgumentNullException(nameof(input), "Input cannot be null or empty.");
151
152 // Validate the format of each key name.
153 foreach (var keyName in keyNames)

156 throw new ArgumentNullException(nameof(keyName), "Key name cannot be null or empty.");
157
158 if (!System.Text.RegularExpressions.Regex.IsMatch(keyName, @"^[a-zA-Z][a-zA-Z0-9\-]{0,126}$"))
159 throw new ArgumentException("Invalid key name. A key name can be 1-127 characters, starting with a letter, containing only 0-9, a-z, A-Z, and -", nameof(keyName));
