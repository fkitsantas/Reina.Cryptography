127 private static void ValidateInput(string input, params string[] keyNames)
128 {
129 // Validate that the input string is not null or empty.
130 if (string.IsNullOrEmpty(input))
131 throw new ArgumentNullException(nameof(input), "Input cannot be null or empty.");
132
133 // Validate the format of each key name.
134 foreach (var keyName in keyNames)
135 {
136 if (string.IsNullOrEmpty(keyName))
137 throw new ArgumentNullException(nameof(keyName), "Key name cannot be null or empty.");
138
139 if (!System.Text.RegularExpressions.Regex.IsMatch(keyName, @"^[a-zA-Z][a-zA-Z0-9\-]{0,126}$"))
140 throw new ArgumentException("Invalid key name. A key name can be 1-127 characters, starting with a letter, containing only 0-9, a-z, A-Z, and -", nameof(keyName));
141 }
142 }
