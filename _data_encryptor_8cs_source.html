79 private byte[] EncryptWithTwofish(byte[] data)
80 {
81 // Initialize Twofish engine and cipher parameters.
82 var twofishEngine = new TwofishEngine();
83 var blockCipher = new CbcBlockCipher(twofishEngine);
84 var cipher = new PaddedBufferedBlockCipher(blockCipher, new Pkcs7Padding());
85 var keyParam = new KeyParameter(_twofishKey);
86 var random = new SecureRandom();
87 var ivTwofish = new byte[twofishEngine.GetBlockSize()];
88 random.NextBytes(ivTwofish);
89 var keyWithIv = new ParametersWithIV(keyParam, ivTwofish);
90 cipher.Init(true, keyWithIv);
91
92 // Perform encryption.
93 var output = new byte[cipher.GetOutputSize(data.Length)];
94 var len = cipher.ProcessBytes(data, 0, data.Length, output, 0);
95 cipher.DoFinal(output, len);
96
97 // Prepend IV to the encrypted data.
98 var twofishEncrypted = new byte[ivTwofish.Length + output.Length];
99 Array.Copy(ivTwofish, 0, twofishEncrypted, 0, ivTwofish.Length);
100 Array.Copy(output, 0, twofishEncrypted, ivTwofish.Length, output.Length);
101
102 return twofishEncrypted;
103 }

142 {
143 // Initialize AES engine.
144 using var aes = Aes.Create();
145 aes.Key = _aesKey;

147
148 // Perform encryption, writing the IV followed by the encrypted data.
149 using var memoryStream = new MemoryStream();
150 memoryStream.Write(aes.IV, 0, aes.IV.Length);
151
152 using var cryptoStream = new CryptoStream(memoryStream, aes.CreateEncryptor(), CryptoStreamMode.Write);
153 cryptoStream.Write(data, 0, data.Length);
