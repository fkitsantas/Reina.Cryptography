191  private byte[] ProcessCipher(IBufferedCipher cipher, byte[] input)
192  {
193  // Calculate the size of the output buffer and perform the decryption.
194  var output = new byte[cipher.GetOutputSize(input.Length)];
195  var len = cipher.ProcessBytes(input, 0, input.Length, output, 0);
196  len += cipher.DoFinal(output, len);
197 
198  // Trim the output to the actual length of the decrypted data.
199  var decrypted = new byte[len];
200  Array.Copy(output, decrypted, len);
