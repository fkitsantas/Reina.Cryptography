63 public byte[] Decrypt(byte[] encryptedBytes)
64 {
65 if (encryptedBytes == null)
66 throw new ArgumentNullException(nameof(encryptedBytes), "Encrypted data cannot be null.");
67
68 // Attempt to decrypt the data in a try-catch block to handle potential cryptographic exceptions.
69 try
70 {
71 // Perform decryption in reverse order of encryption: first AES, then Serpent, and finally Twofish.
72 byte[] aesDecrypted = DecryptWithAES(encryptedBytes);
73 byte[] serpentDecrypted = DecryptWithSerpent(aesDecrypted);
74 byte[] twofishDecrypted = DecryptWithTwofish(serpentDecrypted);
75
76 return twofishDecrypted;
77 }
78 catch (CryptographicException ce)
79 {
80 // Provide more context to the cryptographic exception for better error diagnosis.
81 throw new CryptographicException("Decryption failed: " + ce.Message, ce);
82 }
83 catch (ArgumentException ae)
84 {
85 // Re-throw the argument exception with additional context.
86 throw new ArgumentException("Decryption failed: " + ae.Message, ae);
87 }
88 catch (Exception ex)
89 {
90 // General exception handling as a fallback to catch any other unexpected errors.
91 throw new Exception("Decryption failed: An unexpected error occurred.", ex);
92 }
93 }

101 private byte[] DecryptWithAES(byte[] encryptedBytes)
102 {
103 using var aes = Aes.Create();
104 {
105 int ivLengthAes = aes.BlockSize / 8;
106 if (encryptedBytes.Length < ivLengthAes)
107 throw new ArgumentException("The encrypted data is too short to contain the AES IV.", nameof(encryptedBytes));
108
109 // Extract the IV from the beginning of the encrypted data.
110 var ivAes = new byte[ivLengthAes];
111 Array.Copy(encryptedBytes, ivAes, ivLengthAes);
112
113 aes.Key = _aesKey;
114 aes.IV = ivAes;
115
116 // Decrypt the remaining encrypted data after the IV.
117 using var memoryStream = new MemoryStream(encryptedBytes, ivLengthAes, encryptedBytes.Length - ivLengthAes);
118 using var cryptoStream = new CryptoStream(memoryStream, aes.CreateDecryptor(), CryptoStreamMode.Read);
119 using var streamReader = new MemoryStream();
120 cryptoStream.CopyTo(streamReader);
121 return streamReader.ToArray();
122 }
123 }

131 private byte[] DecryptWithSerpent(byte[] encryptedBytes)
132 {
133 // Initialize Serpent engine and cipher parameters.
134 var serpentEngine = new SerpentEngine();
135 int ivLengthSerpent = serpentEngine.GetBlockSize();
136 if (encryptedBytes.Length < ivLengthSerpent)
137 throw new ArgumentException("The decrypted data is too short to contain the Serpent IV.");
138
139 // Extract the IV from the beginning of the encrypted data.
140 var ivSerpent = new byte[ivLengthSerpent];
141 Array.Copy(encryptedBytes, ivSerpent, ivLengthSerpent);
142
143 var serpentCipher = new PaddedBufferedBlockCipher(new CbcBlockCipher(serpentEngine), new Pkcs7Padding());
144 var serpentKeyParam = new KeyParameter(_serpentKey);
145 var serpentKeyWithIv = new ParametersWithIV(serpentKeyParam, ivSerpent);
146 serpentCipher.Init(false, serpentKeyWithIv);
147
148 // Decrypt the remaining encrypted data after the IV.
149 var serpentInput = new byte[encryptedBytes.Length - ivLengthSerpent];
150 Array.Copy(encryptedBytes, ivLengthSerpent, serpentInput, 0, serpentInput.Length);
151
152 return ProcessCipher(serpentCipher, serpentInput);
153 }

191 private byte[] ProcessCipher(IBufferedCipher cipher, byte[] input)
192 {
193 // Calculate the size of the output buffer and perform the decryption.
194 var output = new byte[cipher.GetOutputSize(input.Length)];
195 var len = cipher.ProcessBytes(input, 0, input.Length, output, 0);
196 len += cipher.DoFinal(output, len);
197
198 // Trim the output to the actual length of the decrypted data.
199 var decrypted = new byte[len];
200 Array.Copy(output, decrypted, len);
201 return decrypted;
202 }
