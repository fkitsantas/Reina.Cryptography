59 public async Task<byte[]> GetEncryptionKeyAsync(string keyName)
60 {
61 // Attempt to retrieve the key from cache.
62 if (_keyCache.TryGetValue(keyName, out var cachedKey))
63 {
64 return cachedKey;
65 }
66
67 try
68 {
69 // Retrieve the key from Azure Key Vault.
70 KeyVaultSecret secret = await _secretClient.GetSecretAsync(keyName);
71 var key = Convert.FromBase64String(secret.Value);
72 _keyCache[keyName] = key;
73 return key;
74 }
75 catch (Azure.RequestFailedException ex) when (ex.Status == 404)
76 {
77 // If the key is not found, generate a new one, store it, and return it.
78 byte[] newKey = Generate256bitKey();
79 await _secretClient.SetSecretAsync(keyName, Convert.ToBase64String(newKey));
80 _keyCache[keyName] = newKey;
81 return newKey;
82 }
83 catch (Azure.RequestFailedException ex) when (ex.Status == 401)
84 {
85 // Handles authentication failures.
86 throw new UnauthorizedAccessException("Failed to authenticate with Azure Key Vault. Please check your credentials.", ex);
87 }
88 catch (Azure.RequestFailedException ex) when (ex.Status == 403)
89 {
90 // Handles authorization failures.
91 throw new UnauthorizedAccessException("Access denied. The application does not have the necessary permissions to access the Azure Key Vault secret.", ex);
92 }
93 catch (Azure.RequestFailedException ex)
94 {
95 // Handles general Azure Key Vault access errors.
96 throw new Exception($"An error occurred while accessing Azure Key Vault: {ex.Message}", ex);
97 }
98 catch (Exception ex)
99 {
100 // Handles all other exceptions.
101 throw new Exception($"An unexpected error occurred: {ex.Message}", ex);
102 }
103 }
